# 명령어
```sh
$ git // 기본명령어 설명
$ git init // git 초기화 명령 (.git 폴더 생성)
$ git config --global user.email "GitHub e-mail" // 사용사 정보 설정 (이메일)
$ git config --global user.name "GitHub name" // 사용사 정보 설정 (이름)
$ git add file_name // 커밋에 추가할 파일을 선택
$ git commit -m "설명추가" // 커밋 상세 설명 추가
$ git log // 지금까지 만든 커밋 확인
$ git checkout "7자리 커밋 아이디" // 되돌리려는 커밋아이디 선택하여 해당 커밋으로 되돌림
$ git checkout - // 최신 커밋으로 되돌아감
$ git remote add origin https://github.com/Son-sk/git_use_manual.git  // 원격저장소 선택 [github주소]
$ git push origin master // 로컬저장소에 커밋을 원격저장소(Github)에 업로드 (push)
$ git pull origin master // 원격저장소(Github)에 새로운 커밋을 로컬저장소로 업데이트 (Pull)
$ git clone https://github.com/Son-sk/git_use_manual.git . // 원격저장소(Git Hub)에서 로컬 저장소(내컴퓨터)로 다운로드 
```
